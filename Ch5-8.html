backupServ

Windows 使用时间久了，运行会变的很慢，让人很不爽。 除了删除垃圾文件、优化注册表这些操作之外，系统服务的修改也是很重要的。 可是系统管理员不是满天下都有的，系统服务的名字又不好记，第三方的软件很多也不能提供备份。 为此写了这个脚本，里面涵盖了备份、优化常见版本 Windows 的服务。

@echo off
:menu
cls
title Services management script
color 0a
echo.
echo.
echo. 1. Backup all services' state
echo. 2. Restore service backup
echo. 3. Quit
echo.
echo.update:
echo. 4. Windows XP default services configuration
echo. 5. Windows XP optimized services configuration
echo. 6. Windows Server 2003 optimized services configuration
echo. 7. Windows 7 default services configuration ^(experimental^)
echo. 8. Super lite Windows services mode ^(Advanced user only^)
echo.
echo.
set choice=
set /p choice=Input your choice here:  
if .%choice%==.1 goto BakServ
if .%choice%==.2 goto Restore
if .%choice%==.3 goto eof
set serv=
if .%choice%==.6 call Serv23.bat & goto menu
if .%choice%==.7 call Serv7.bat & goto menu
if .%choice%==.4 set serv=default
if .%choice%==.5 set serv=optixp
if .%choice%==.8 set serv=lite
if defined serv call Servxp.bat & goto menu
color 0e
echo.
echo. Sorry, wrong choice! & ping 127.1 >nul & goto menu
:BakServ
cls
for /f "tokens=1,2 delims=: " %%i in ('time /t') do set bakname=-%%i-%%j
set bakname=Serv%date:~5,2%-%date:~8,2%%bakname%
echo.
echo. Let's backup all services
echo. the bak file store into %userprofile%\ 
echo. and named "%bakname%"
echo.
echo. 
echo. press ENTER to get backup . . .
set /p backup=if you don't like the name, you can type a new name here:  
if defined backup set bakname=%backup%
set backup=
set bakname=%bakname%.bat
pushd %tmp%
sc query type= service state= all | find "SERVICE_NAME" > serv.name
if exist serv.list del /q serv.list
cls
echo.
echo.
echo. NEO's Edition is Okay~
echo.
echo.    Loading . . .
echo.
for /f "tokens=2" %%i in (serv.name) do (
        for /f "tokens=4,5 delims=:_ " %%j in ('sc qc "%%i" ^| find "START_TYPE"') do (
                echo %%i %%j %%k >> serv.list
        )
)
if exist serv.list.db del /q serv.list.db > nul
for /f "tokens=1,2,3" %%i in (serv.list) do (
        if "%%j"=="DEMAND" (
                for /f "tokens=4" %%a in ('sc query "%%i" ^| find "STATE"') do ( echo %%i %%j %%a >> serv.list.db & echo Service: %%i is %%j now ^(%%a^) )
        ) else ( echo %%i %%j %%k >> serv.list.db & echo Service: %%i is %%j now ^(%%k.^) )
        
)
echo.title Restore all services > "%bakname%"
echo.echo off >> "%bakname%"
echo.echo.>> "%bakname%"
echo.echo. please wait for me a while >> "%bakname%"
echo.echo. loading . . . >> "%bakname%"
echo.ping 127.1 -n 2 ^> nul >> "%bakname%"
echo.echo. >> "%bakname%"
for /f "tokens=1,2,3" %%i in (serv.list.db) do (
        if "%%j"=="DISABLED" echo net STOP %%i 2^> nul >> "%bakname%"
        echo sc config %%i start= %%j >> "%bakname%"
        if "%%j"=="AUTO" echo net START %%i 2^> nul >> "%bakname%"
        if "%%k"=="RUNNING" echo net START %%i 2^> nul >> "%bakname%"
        if "%%k"=="STOPPED" echo net STOP %%i 2^> nul >> "%bakname%"
)
echo color 0e >> "%bakname%"
echo ping 127.1 -n 2 ^> nul >> "%bakname%"
echo color 0a >> "%bakname%"
echo echo. >> "%bakname%"
echo echo. Done! >> "%bakname%"
echo ping 127.1 -n 2 ^> nul >> "%bakname%"
echo. >> "%bakname%"
color 0e
echo. Cleaning up . . .
del /q serv.* > nul
move /y "%bakname%" "%userprofile%\" > nul
ping 127.1 -n 1 > nul
popd
color 0a
echo. Done, press any key to continue . . .
pause > nul
goto menu
:Restore
pushd %userprofile%
dir Serv??-??-??-??.bat > nul 2> nul
if errorlevel 1 goto seemore
if exist bak.list del /q bak.list > nul
for /f "tokens=3*" %%i in ('dir Serv??-??-??-??.bat ^| find /i ".bat"') do ( echo %%j >> bak.list )
echo.
echo. That's what I found:
echo.
type bak.list | more
echo. If you want to see more script files, press ENTER to continue . . .
set bakfile=
set /p bakfile=Which one to restore:  
echo.
if not defined bakfile goto seemore
if exist "%bakfile%" call "%bakfile%" &popd& goto menu
:seemore
if exist more.list del /q more.list > nul
for /f "tokens=3*" %%i in ('dir ^| find /i ".bat" ^| findstr /v /i "Serv[[0-9]*-]*"') do ( echo %%j >> more.list )
if exist more.list goto showBat
color 0b
echo. Sorry, no script file found & goto whereBat
:showBat
echo. Maybe you have modified the file's name
echo. here is a list of all scripts on your home path:
echo.
type more.list | more
:whereBat
echo.
echo. press ENTER to back the main menu . . .
if exist bak.list del /q bak.list > nul
if exist more.list del /q more.list > nul
set /p bakpath=Or, input the script name listed here:  
if not defined bakpath popd&goto menu
if exist "%bakpath%" call "%bakpath%" &popd& goto menu
color 0e
echo.
echo. Sorry, script file doesn't exist! & ping 127.1 > nul &popd& goto menu
:eof
set choice=
set serv=
set bakfile=
set bakpath=
set bakname=
echo.
echo. bye 
echo. Yours, Neo & ping 127.1 -n 2 > nul
